  23:     /**
  24:      * Process checkout from localStorage cart items.
  25:      * Expects JSON body: { items: [{ id, name, price, qty }] }
  26:      */
  27:     public function checkout(Request $request)
  28:     {
  29:         $user = $request->user();
  30:         $address = $this->resolveProfileAddress($user);
  31:         if (blank($user?->name) || blank($user?->email) || blank($user?->phone) || blank($address)) {
  32:             return back()->withErrors([
  33:                 'cart' => 'Lengkapi data profil (nama, email, no. HP, alamat) sebelum checkout.',
  34:             ]);
  35:         }
  36: 
  37:         $data = $request->validate([
  38:             'items'         => 'required|array|min:1',
  39:             'items.*.id'    => 'required|integer|exists:products,id',
  40:             'items.*.qty'   => 'required|integer|min:1',
  41:             'notes'         => 'nullable|string|max:500',
  42:         ]);
  43: 
  44:         $sessionAffiliateId = (int) $request->session()->get('ref_affiliate_id', 0);
  45:         $affiliateId = null;
  46:         $isGuestUser = (bool) ($user && $user->hasRole('guest'));
  47: 
  48:         if ($user && $user->hasRole('affiliate')) {
  49:             $affiliateId = $user->affiliate?->id;
  50:         }
  51: 
  52:         if ($isGuestUser && $sessionAffiliateId > 0) {
  53:             $sessionAffiliate = Affiliate::where('id', $sessionAffiliateId)
  54:                 ->where('is_active', true)
  55:                 ->value('id');
  56:             if (!$affiliateId && $sessionAffiliate) {
  57:                 $affiliateId = $sessionAffiliate;
  58:             }
  59:         }
  60: 
  61:         if (!$affiliateId && $isGuestUser) {
  62:             $randomAffiliate = app(\App\Services\AffiliateService::class)->assignRandomSponsor();
  63:             if ($randomAffiliate) {
  64:                 $affiliateId = $randomAffiliate->id;
  65:                 $request->session()->put('ref_affiliate_id', $randomAffiliate->id);
  66:                 $request->session()->put('ref_affiliate_slug', $randomAffiliate->slug);
  67:             }
  68:         }
  69: 
  70:         try {
  71:             $order = DB::transaction(function () use ($data, $user, $affiliateId, $address) {
  72:                 $productIds = collect($data['items'])->pluck('id');
  73:                 $products   = Product::whereIn('id', $productIds)
  74:                     ->where('is_active', true)
  75:                     ->lockForUpdate()
  76:                     ->get()
  77:                     ->keyBy('id');
  78: 
  79:                 $grandTotal            = 0;
  80:                 $totalQty              = 0;
  81:                 $firstProduct          = null;
  82:                 $generatesActivationCode = false;
  83:                 $lineItems             = [];
  84: 
  85:                 foreach ($data['items'] as $cartItem) {
  86:                     $product = $products->get($cartItem['id']);
  87: 
  88:                     if (!$product) {
  89:                         throw new \RuntimeException("Produk ID {$cartItem['id']} tidak ditemukan atau tidak aktif.");
  90:                     }
  91: 
  92:                     $qty = (int) $cartItem['qty'];
  93: 
  94:                     if ((int) $product->stock < $qty) {
  95:                         throw new \RuntimeException("Stok produk \"{$product->name}\" tidak mencukupi.");
  96:                     }
  97: 
  98:                     $lineTotal   = (float) $product->harga_akhir * $qty;
  99:                     $grandTotal += $lineTotal;
 100:                     $totalQty   += $qty;
 101: 
 102:                     $generatesActivationCode = $generatesActivationCode
 103:                         || (bool) $product->generates_activation_code
 104:                         || strtolower((string) $product->type) === 'bundle';
 105: 
 106:                     if ($firstProduct === null) {
 107:                         $firstProduct = $product;
 108:                     }
 109: 
 110:                     $lineItems[] = [
 111:                         'product'    => $product,
 112:                         'qty'        => $qty,
 113:                         'harga_awal' => (float) $product->harga_awal,
 114:                         'diskon'     => (float) $product->diskon,
 115:                         'harga_akhir'=> (float) $product->harga_akhir,
 116:                     ];
 117:                 }
 118: 
 119:                 $this->enforceGuestBundleRule($user, $lineItems);
 120: 
 121:                 $order = Order::create([
 122:                     'order_number'             => 'ORD-' . uniqid(),
 123:                     'user_id'                  => $user->id,
 124:                     'affiliate_id'             => $affiliateId,
 125:                     'payment_method'           => 'midtrans_snap',
 126:                     'midtrans_order_id'        => 'MT-' . uniqid(),
 127:                     'shipping_data'            => [
 128:                         'recipient_name'  => $user->name,
 129:                         'recipient_phone' => $user->phone,
 130:                         'address'         => $address,
 131:                     ],
 132:                     'product_type'             => strtolower((string) ($firstProduct?->type ?? 'single')),
 133:                     'product_id'               => $firstProduct?->id,
 134:                     'product_name'             => $firstProduct?->name ?? 'Cart Checkout',
 135:                     'quantity'                 => $totalQty,
 136:                     'price'                    => count($lineItems) === 1
 137:                         ? $lineItems[0]['harga_akhir']
 138:                         : ($totalQty > 0 ? $grandTotal / $totalQty : 0),
 139:                     'total_amount'             => $grandTotal,
 140:                     'shipping_cost'            => 0,
 141:                     'tax_amount'               => 0,
 142:                     'grand_total'              => $grandTotal,
 143:                     'payment_status'           => 'pending',
 144:                     'status'                   => 'pending',
 145:                     'generates_activation_code'=> $generatesActivationCode,
 146:                     'notes'                    => $data['notes'] ?? 'Created from public cart checkout',
 147:                 ]);
 148: 
 149:                 foreach ($lineItems as $line) {
 150:                     $product = $line['product'];
 151: 
 152:                     OrderItem::create([
 153:                         'order_id'              => $order->id,
 154:                         'product_id'            => $product->id,
 155:                         'package_id'            => null,
 156:                         'quantity'              => $line['qty'],
 157:                         'gives_activation_code' => (bool) ($product->generates_activation_code ?? false)
 158:                             || strtolower((string) ($product->type ?? '')) === 'bundle',
 159:                         'harga_awal'            => $line['harga_awal'],
 160:                         'diskon'                => $line['diskon'],
 161:                         'harga_akhir'           => $line['harga_akhir'],
 162:                     ]);
 163: 
 164:                     $product->decrement('stock', $line['qty']);
 165:                 }
 166: 
 167:                 return $order->load('items.product');
 168:             });
 169:         } catch (\Throwable $e) {
 170:             return back()->withErrors(['cart' => 'Checkout gagal: ' . $e->getMessage()]);
 171:         }
 172: 
 173:         try {
 174:             $redirectUrl = $this->createMidtransTransaction($order, $user);
 175:         } catch (\Throwable $e) {
 176:             Log::error('Public cart midtrans failed: ' . $e->getMessage(), ['order_id' => $order->id]);
 177: 
 178:             app(OrderService::class)->cancelPendingOrder($order->id, 'Midtrans create transaction failed');
 179: 
 180:             return back()->withErrors(['cart' => 'Gagal membuat transaksi Midtrans.']);
 181:         }
 182: 
 183:         return back()->with('midtrans', [
 184:             'snap_token'   => $order->midtrans_data['snap_token'] ?? null,
 185:             'redirect_url' => $redirectUrl,
 186:             'order_number' => $order->order_number,
 187:         ]);
 188:     }

 191:      * Cancel the latest pending Midtrans order for this user.
 192:      */
 193:     public function cancel(Request $request)
 194:     {
 195:         $user = $request->user();
 196: 
 197:         $pendingOrder = Order::where('user_id', $user->id)
 198:             ->where('payment_method', 'midtrans_snap')
 199:             ->where('payment_status', 'pending')
 200:             ->where('status', 'pending')
 201:             ->latest('id')
 202:             ->first();
 203: 
 204:         if ($pendingOrder) {
 205:             app(OrderService::class)->cancelPendingOrder(
 206:                 $pendingOrder->id,
 207:                 'Cancelled by user from public cart'
 208:             );
 209:         }
 210: 
 211:         return back()->with('success', 'Pesanan dibatalkan.');
 212:     }

 216:     private function createMidtransTransaction(Order $order, $user): string
 217:     {
 218:         MidtransConfig::$serverKey   = config('midtans.server_key') ?: env('MIDTRANS_SERVER_KEY');
 219:         MidtransConfig::$isProduction = (bool) (config('midtans.is_production') ?? env('MIDTRANS_IS_PRODUCTION', false));
 220:         MidtransConfig::$isSanitized  = (bool) (config('midtans.is_sanitized') ?? env('MIDTRANS_IS_SANITIZED', true));
 221:         MidtransConfig::$is3ds        = (bool) (config('midtans.is_3ds') ?? env('MIDTRANS_IS_3DS', true));
 222: 
 223:         $itemDetails = $order->items->map(function ($item) {
 224:             return [
 225:                 'id'       => (string) ($item->product_id ?? $item->id),
 226:                 'price'    => (int) round((float) $item->harga_akhir),
 227:                 'quantity' => (int) $item->quantity,
 228:                 'name'     => substr($item->product?->name ?? 'Product', 0, 50),
 229:             ];
 230:         })->values()->all();
 231: 
 232:         $params = [
 233:             'transaction_details' => [
 234:                 'order_id'     => $order->midtrans_order_id,
 235:                 'gross_amount' => (int) round((float) $order->grand_total),
 236:             ],
 237:             'item_details'     => $itemDetails,
 238:             'customer_details' => [
 239:                 'first_name' => $user->name ?? 'Customer',
 240:                 'email'      => $user->email ?? 'customer@example.com',
 241:                 'phone'      => $user->phone ?? '-',
 242:             ],
 243:             'callbacks' => [
 244:                 'finish'  => route('cart'),
 245:                 'error'   => route('cart'),
 246:                 'pending' => route('cart'),
 247:             ],
 248:         ];
 249: 
 250:         $transaction = Snap::createTransaction($params);
 251: 
 252:         $order->update([
 253:             'midtrans_data'      => [
 254:                 'snap_token'   => $transaction->token ?? null,
 255:                 'redirect_url' => $transaction->redirect_url ?? null,
 256:                 'source'       => 'public_cart',
 257:             ],
 258:             'payment_reference'  => $transaction->token ?? null,
 259:         ]);
 260: 
 261:         return $transaction->redirect_url;
 262:     }

 264:     private function resolveProfileAddress($user): ?string
 265:     {
 266:         $address = $user->profile?->address;
 267:         if (is_string($address) && trim($address) !== '') {
 268:             return trim($address);
 269:         }
 270:         if (!is_array($address)) {
 271:             return null;
 272:         }
 273:         if (!empty($address['full_address'])) {
 274:             return (string) $address['full_address'];
 275:         }
 276:         $parts = array_filter([
 277:             $address['street']      ?? null,
 278:             $address['subdistrict'] ?? null,
 279:             $address['district']    ?? null,
 280:             $address['city']        ?? null,
 281:             $address['province']    ?? null,
 282:             $address['postal_code'] ?? null,
 283:         ]);
 284:         return !empty($parts) ? implode(', ', $parts) : null;
 285:     }

 287:     private function enforceGuestBundleRule($user, array $lineItems): void
 288:     {
 289:         if (!$user || !$user->hasRole('guest')) {
 290:             return;
 291:         }
 292: 
 293:         $bundleQty = collect($lineItems)->sum(function (array $line): int {
 294:             $type = strtolower((string) ($line['product']->type ?? ''));
 295:             return $type === 'bundle' ? (int) ($line['qty'] ?? 0) : 0;
 296:         });
 297: 
 298:         if ($bundleQty <= 0) {
 299:             return;
 300:         }
 301: 
 302:         if ($bundleQty > 1) {
 303:             throw new \RuntimeException('Guest hanya bisa membeli 1 produk bundle per transaksi.');
 304:         }
 305: 
 306:         $hasPaidBundle = Order::query()
 307:             ->where('user_id', $user->id)
 308:             ->where('payment_status', 'paid')
 309:             ->whereHas('items.product', function ($q) {
 310:                 $q->whereRaw('LOWER(type) = ?', ['bundle']);
 311:             })
 312:             ->exists();
 313: 
 314:         if ($hasPaidBundle) {
 315:             throw new \RuntimeException('Guest sudah pernah membeli produk bundle sebelumnya.');
 316:         }
 317:     }
