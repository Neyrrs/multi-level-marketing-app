 199:     /**
 200:      * Apply commission untuk order
 201:      * Follow commission method & rules yang sudah di-setup di database
 202:      * 
 203:      * @param Order $order
 204:      * @param CommissionMethod|null $method - Jika null, gunakan default
 205:      */
 206:     public function applyCommission(
 207:         Order $order,
 208:         ?CommissionMethod $method = null
 209:     ): array {
 210:         $commissions = [];
 211: 
 212:         // Get commission method (default: Direct Commission)
 213:         if (!$method) {
 214:             $method = CommissionMethod::where('name', 'Direct Commission - Percentage')
 215:                 ->where('is_active', true)
 216:                 ->first();
 217:         }
 218: 
 219:         if (!$method || !$method->is_active) {
 220:             return $commissions;
 221:         }
 222: 
 223:         // Get rules ordered by priority
 224:         $rules = $method->rules()
 225:             ->where('is_active', true)
 226:             ->orderBy('priority')
 227:             ->get();
 228: 
 229:         // Get seller affiliate
 230:         $currentAffiliate = $order->affiliate;
 231:         if (!$currentAffiliate) {
 232:             return $commissions;
 233:         }
 234: 
 235:         // Loop through levels & apply commission
 236:         $level = 1;
 237:         $tempAffiliate = $currentAffiliate;
 238: 
 239:         while ($tempAffiliate && $level <= 5) {  // Max 5 levels deep
 240:             // Find rule by priority matching level
 241:             $rule = $rules->where('priority', $level)->first();
 242:             if (!$rule) {
 243:                 break;
 244:             }
 245: 
 246:             // Calculate commission
 247:             $commissionAmount = $this->calculateCommissionAmount(
 248:                 $order->total_amount,
 249:                 $rule
 250:             );
 251: 
 252:             if ($commissionAmount <= 0) {
 253:                 break;
 254:             }
 255: 
 256:                 // Delegate creation to CommissionService
 257:                 $commissionService = new CommissionService();
 258:                 $commission = $commissionService->distributeCommission(
 259:                     $tempAffiliate->id,
 260:                     $commissionAmount,
 261:                     $method->id,
 262:                     $rule->id,
 263:                     'level',
 264:                     $order->id,
 265:                     $level
 266:                 );
 267: 
 268:                 $commissions[] = $commission;
 269: 
 270:             // Move to next level (parent affiliate)
 271:             $tempAffiliate = Affiliate::where('user_id', $tempAffiliate->upline_id)
 272:                 ->first();
 273:             $level++;
 274:         }
 275: 
 276:         return $commissions;
 277:     }

 367:     /**

 370:     private function calculateCommissionAmount(float $orderAmount, $rule): float
 371:     {
 372:         // Check minimum sales requirement
 373:         if ($rule->min_sales && $orderAmount < $rule->min_sales) {
 374:             return 0;
 375:         }
 376: 
 377:         // Check maximum sales limit
 378:         if ($rule->max_sales && $orderAmount > $rule->max_sales) {
 379:             return 0;
 380:         }

 382:         if ($rule->percentage) {
 383:             return ($orderAmount * $rule->percentage) / 100;
 384:         } elseif ($rule->fixed_amount) {
 385:             return $rule->fixed_amount;
 386:         }
 387: 
 388:         return 0;

 493:     /**
 494:      * Confirm a pending affiliate and place them into the tree.
 495:      */
 496:     public function confirmAffiliate(int $affiliateId, string $position = 'left', ?Affiliate $placementUpline = null): Affiliate
 497:     {
 498:         $affiliate = Affiliate::findOrFail($affiliateId);
 499:         if ($affiliate->is_active) {
 500:             return $affiliate; // already active
 501:         }
 502: 
 503:         $sponsor = Affiliate::where('user_id', $affiliate->sponsor_id)->first();
 504:         $targetUpline = $placementUpline && $sponsor
 505:             && ((int) $placementUpline->sponsor_id === (int) $sponsor->user_id || (int) $placementUpline->id === (int) $sponsor->id)
 506:             ? $placementUpline
 507:             : $sponsor;
 508: 
 509:         // set position and active
 510:         $defaultPlanId = $this->resolveDefaultPlanId();
 511:         $affiliate->update([
 512:             'position' => in_array($position, ['left','right']) ? $position : 'none',
 513:             'upline_id' => $targetUpline?->user_id,
 514:             'level' => $targetUpline ? ($targetUpline->level + 1) : $affiliate->level,
 515:             'is_active' => true,
 516:             'activated_at' => now(),
 517:             'active_until' => now()->addMonth(),
 518:             'commission_plan_id' => $affiliate->commission_plan_id ?: $defaultPlanId,
 519:         ]);
 520: 
 521:         // Promote user role to affiliate only after approval.
 522:         $affiliate->user?->syncRoles(['affiliate']);
 523: 
 524:         // create mlm tree node under selected upline tree
 525:         $parentTree = $targetUpline?->mlmTree ?? null;
 526:         $parentId = $targetUpline?->id ?? null;
 527: 
 528:         $node = MlmTree::create([
 529:             'affiliate_id' => $affiliate->id,
 530:             'parent_id' => $parentId,
 531:             'position' => $affiliate->position,
 532:             'depth' => $affiliate->level ?? 1,
 533:             'path' => null,
 534:         ]);
 535: 
 536:         // recalc nested set
 537:         $this->updateNestedSet($affiliate->id);
 538: 
 539:         return $affiliate;
 540:     }

 542:     /**
 543:      * Recalculate nested set positions for MLM tree.
 544:      * This is a recursive DFS assigner starting from root nodes.
 545:      */
 546:     public function updateNestedSet(?int $changedAffiliateId = null): void
 547:     {
 548:         // start from all root nodes (parent_id is null)
 549:         $counter = 1;
 550: 
 551:         $roots = MlmTree::whereNull('parent_id')->orderBy('id')->get();
 552: 
 553:         $recurse = function ($node) use (&$recurse, &$counter) {
 554:             $left = $counter++;
 555:             $children = MlmTree::where('parent_id', $node->affiliate_id)->orderBy('id')->get();
 556:             foreach ($children as $child) {
 557:                 $recurse($child);
 558:             }
 559:             $right = $counter++;
 560: 
 561:             $node->update(['left_position' => $left, 'right_position' => $right]);
 562:         };
 563: 
 564:         foreach ($roots as $root) {
 565:             $recurse($root);
 566:         }
 567:     }
