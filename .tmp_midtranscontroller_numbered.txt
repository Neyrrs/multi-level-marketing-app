  62:                 || ($transactionStatus === 'capture' && ($fraudStatus === '' || $fraudStatus === 'accept'));

  66:                 if ($processed) {
  67:                     return response()->json(['ok']);

  71:                 return response()->json(['error' => 'processing failed'], 500);

  82:

  85:         } catch (\Throwable $e) {

  87:             // Respond 200 to avoid repeated retries, but surface logs
